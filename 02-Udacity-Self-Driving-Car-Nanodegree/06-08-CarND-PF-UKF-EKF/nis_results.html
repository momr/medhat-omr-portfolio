def plot_nis(df, title, target_val):
    x = df['Timestamp']
    y = df['NIS']
    horiz_line_data = np.array([target_val for i in range(len(x))])
    # plot
    plt.figure()
    plt.plot(x, y)
    plt.plot(x, horiz_line_data)
    plt.title(title)
    plt.show()

def calc_nis(df, target_val):
    nis = df['NIS']
    result = [n for n in nis if n < target_val]
    
    return len(result)/len(nis)*100
